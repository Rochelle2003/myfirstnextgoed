# Supabase Configuration
# Kopieer dit bestand naar .env.local en vul je eigen waarden in

# Je Supabase project URL (vind je in Settings > API)

# Je Supabase anon/public key (vind je in Settings > API)
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key-here
